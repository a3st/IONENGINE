34inline std::basic_string<T, std::char_traits<T>, std::allocator<T>> format(const std::basic_string<T, std::char_traits<T>, std::allocator<T>>& fmt, const I& arg, A&&... args) {
35 auto it_arg = std::find(fmt.begin(), fmt.end(), '{');
36 auto it_type = std::next(it_arg);
37
38 if (it_arg == fmt.end() || it_type == fmt.end()) {
39 return format(fmt);
40 }
41
42 std::basic_stringstream<T, std::char_traits<T>, std::allocator<T>> ss;
43 ss << std::basic_string<T, std::char_traits<T>, std::allocator<T>>(fmt.begin(), it_arg);
44
45 switch (*it_type) {
46 case '}': {
47 ss << arg << std::basic_string<T, std::char_traits<T>, std::allocator<T>>(it_type + 1, fmt.end());
48 break;
49 }
50 case 'x': {
51 if constexpr (std::is_integral<I>::value) {
52 auto it_optional = std::next(it_type);
53 if (*it_optional == ':') {
54 ++it_optional;
55 if (*it_optional != '}') {
56 ss <<
57 std::hex <<
58 std::setw(static_cast<std::streamsize>(*it_optional - 48)) <<
59 std::setfill('0') << static_cast<unsigned>(arg) <<
60 std::dec;
61
62 if (*std::next(it_optional) != '}') {
63 ss << std::basic_string<T, std::char_traits<T>, std::allocator<T>>(it_optional + 3, fmt.end());
64 } else {
65 ss << std::basic_string<T, std::char_traits<T>, std::allocator<T>>(it_optional + 2, fmt.end());
66 }
67 }
68 }
69 } else {
70 ss << arg << std::basic_string<T, std::char_traits<T>, std::allocator<T>>(it_type + 1, fmt.end());
71 }
72 break;
73 }
74 }
75 return internal::format(ss.str(), std::forward<A>(args)...);
76}
