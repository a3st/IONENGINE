10class InputSystem {
11public:

13 InputSystem() {

15 m_previous_keys.fill(false);

18 }

20 void tick() {
21
22 m_previous_keys = m_current_keys;
23 m_current_axis.fill(0.0f);
24 }
25
26 void on_event_handle(const platform::WindowEventHandler& event) {
27 switch(event.event_type) {
28 case platform::WindowEvent::KeyboardInput: {
29 auto input = std::get<platform::KeyboardInput>(event.event);
30 m_current_keys[input.scan_code] = static_cast<bool>(input.state);
31 break;
32 }
33 case platform::WindowEvent::MouseInput: {
34 auto input = std::get<platform::MouseInput>(event.event);
35 switch(input.button) {
36 case platform::MouseButton::Left: m_current_keys[static_cast<uint32>(KeyCode::MouseLeft)] = static_cast<bool>(input.state); break;
37 case platform::MouseButton::Right: m_current_keys[static_cast<uint32>(KeyCode::MouseRight)] = static_cast<bool>(input.state); break;
38 case platform::MouseButton::Middle: m_current_keys[static_cast<uint32>(KeyCode::MouseMiddle)] = static_cast<bool>(input.state); break;
39 case platform::MouseButton::Four: m_current_keys[static_cast<uint32>(KeyCode::MouseFour)] = static_cast<bool>(input.state); break;
40 case platform::MouseButton::Five: m_current_keys[static_cast<uint32>(KeyCode::MouseFive)] = static_cast<bool>(input.state); break;
41 }
42 break;
43 }
44 case platform::WindowEvent::MouseMoved: {
45 auto input = std::get<platform::MouseMoved>(event.event);
46 if(input.relative) {
47 m_current_axis[static_cast<uint32>(AxisCode::MouseX)] = static_cast<float>(input.position.x);
48 m_current_axis[static_cast<uint32>(AxisCode::MouseY)] = static_cast<float>(input.position.y);
49 }
50 break;
51 }
52 default: break;
53 }
54 }
55
56 bool get_key_down(const KeyCode key_code) const { return m_current_keys[static_cast<uint32>(key_code)] && !m_previous_keys[static_cast<uint32>(key_code)]; }
57 bool get_key_up(const KeyCode key_code) const { return !m_current_keys[static_cast<uint32>(key_code)] && m_previous_keys[static_cast<uint32>(key_code)]; }
58 bool get_key(const KeyCode key_code) const { return m_current_keys[static_cast<uint32>(key_code)]; }
59
60private:
61
62 std::array<bool, key_codes_max> m_current_keys;
63 std::array<bool, key_codes_max> m_previous_keys;
64 std::array<float, axis_codes_max> m_current_axis;
65};
