47 Matrix<T> to_matrix() const {
48 T sqw = w * w;
49 T sqx = x * x;
50 T sqy = y * y;
51 T sqz = z * z;
52 T inverse = 1 / (sqx + sqy + sqz + sqw);
53
54 Matrix<T> mat = Matrix<T>::identity();
55 mat.m00 = (sqx - sqy - sqz + sqw) * inverse;
56 mat.m11 = (-sqx + sqy - sqz + sqw) * inverse;
57 mat.m22 = (-sqx - sqy + sqz + sqw) * inverse;
58 mat.m10 = 2 * (x * y + z * w) * inverse;
59 mat.m01 = 2 * (x * y - z * w) * inverse;
60 mat.m20 = 2 * (x * z - y * w) * inverse;
61 mat.m02 = 2 * (x * z + y * w) * inverse;
62 mat.m21 = 2 * (y * z + x * w) * inverse;
63 mat.m12 = 2 * (y * z - x * w) * inverse;
64 return mat;
65 }

71 void to_angle_axis(T* angle, Vector3<T>* axis) const {
72 Quaternion quat;
73 quat = *this;
74
75 if (w > 1) {
76 quat.normalize();
77 }
78
79 *angle = static_cast<T>(2 * math::acos(quat.w) * math::rad2deg);
80 T s = math::sqrt(1 - quat.w * quat.w);
81
82 if (s < 0.001) {
83 axis->x = quat.x;
84 axis->y = quat.y;
85 axis->z = quat.z;
86 } else {
87 axis->x = quat.x / s;
88 axis->y = quat.y / s;
89 axis->z = quat.z / s;
90 }

122 static Quaternion euler(const T x, const T y, const T z) {
123 T rotx, roty, rotz;
124 rotx = static_cast<T>(x * math::deg2Rad);
125 roty = static_cast<T>(y * math::deg2Rad);
126 rotz = static_cast<T>(z * math::deg2Rad);
127
128 T sinx, siny, sinz, cosx, cosy, cosz;
129 sinx = math::sin(rotx / 2);
130 siny = math::sin(roty / 2);
131 sinz = math::sin(rotz / 2);
132 cosx = math::cos(rotx / 2);
133 cosy = math::cos(roty / 2);
134 cosz = math::cos(rotz / 2);
135
136 Quaternion quat;
137 quat.w = cosx * cosy * cosz + sinx * siny * sinz;
138 quat.x = sinx * cosy * cosz + cosx * siny * sinz;
139 quat.y = cosx * siny * cosz - sinx * cosy * sinz;
140 quat.z = cosx * cosy * sinz - sinx * siny * cosz;
141 return quat;
142 }

144 static Quaternion angle_axis(const T angle, const Vector3<T>& axis) {
145 T rot_angle = static_cast<T>(angle * math::deg2rad);
146 T rot_sin = math::sin(rot_angle / 2);
147 Vector3<T> norm_axis = axis.normalize();
148 return { norm_axis.x * rot_sin, norm_axis.y * rot_sin, norm_axis.z * rot_sin, math::cos(rot_angle / 2) };
149 }
