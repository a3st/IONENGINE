20class Lexer {
21public:
22
23 Lexer(const std::filesystem::path& file_path) {
24
25 std::ifstream ifs(file_path, std::ios::end);
26 if(!ifs.is_open()) {
27 throw std::runtime_error("Error loading a file");
28 }
29
30 std::streampos position = ifs.tellg();
31 ifs.seekg(std::ios::beg);
32 ifs.read(reinterpret_cast<char*>(m_buffer.data()), position);
33 }
34
35 void tokenize() {
36
37 for(uint32 i = 0; i < m_buffer.size(); ++i) {
38
39 switch(m_buffer[i]) {
40 case ';': m_tokens.emplace(Token { Token::Type::Commentary, i }); break;
41 case '[': m_tokens.emplace(Token { Token::Type::SectionOpen, i }); break;
42 case ']': m_tokens.emplace(Token { Token::Type::SectionClose, i }); break;
43 case '=': m_tokens.emplace(Token { Token::Type::Equal, i }); break;
44 }
45 }
46 }
47
48 Token get_current() const { return m_tokens.top(); }
49 std::optional<Token> get_next() {
50 if(!m_tokens.empty()) {
51 m_tokens.pop();
52 return std::optional<Token>(m_tokens.top());
53 } else {
54 return std::nullopt;
55 }
56 }
57
58private:
59
60 std::string m_buffer;
61 std::stack<Token> m_tokens;
62};
